<script type="text/javascript">
    $(document).ready(function () {

        /**
         * start check list form
         */
        $("#checklist_form").validate({

            rules: {
                check_title: {
                    required: true,
//                    letter_with_whitespace: true
                }
            },
            messages: {
                check_title: {
                    required: "Please Enter Check List Name",
                    letter_with_whitespace: "Only letter and whitespace allowed"
                }
            },
            submitHandler: function (form) {
                var url = "<?= base_url("hr_checklist_master/make_list") ?>";

                var json = '{}';
                var object = JSON.parse(json);

                for (var i = 0; i < form.length - 1; i++) {
                    if (form[i].value !== "") {
                        object[form[i].name] = window.btoa(form[i].value);
                    }
                }

                $.ajax({
                    url: url,
                    data: object,
                    method: "post"
                }).done(function (success) {
                    console.log(success);
                    $('#checklistModal').modal('toggle');
                    success = JSON.parse(success);
                    var message = success["response"];
                    console.log(message);
                    $("#checklist_form").trigger("reset");
                    load_table();
                    $.notific8(message, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000,
                    });

                }).fail(function (error) {
                    console.log(error);
                    $('#checklistModal').modal('toggle');
                    $.notific8(success, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000,
                        theme: 'ruby',
                    });
                });
            }
        });

        /*
         * end check list form
         */

        /*
         *
         * check list option form
         */
        var setlist_ref;
        $('#checklist_option_form').on('submit', function (event) {
            event.preventDefault();
            $.ajax({
                url: "<?= base_url("hr_checklist_master/make_list_option") ?>",
                type: "POST",
                data: new FormData(this),
                contentType: false,
                cache: false,
                processData: false,
                success: function (success) {
                    success = JSON.parse(success);
                    var message = success["response"];
                    $("#checklist_option_form").trigger("reset");
                    $.notific8(message, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000
                    });
                    load_checklist_table(setlist_ref)
                },
                error: function (error) {
                    console.log(error);
                    error = JSON.parse(error);
                    var message = error["response"];
                    console.log(message);
                    $("#checklist_option_form").trigger("reset");
                    $.notific8(message, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000
                    });
                }
            });
        }
        );

        /*
         *
         * Asign checklist form
         */
        $('#assign_check_to_user_form').on('submit', function (event) {
            event.preventDefault();
            $.ajax({
                url: "<?= base_url("hr_checklist_master/assign_checklist") ?>",
                type: "POST",
                data: new FormData(this),
                contentType: false,
                cache: false,
                processData: false,
                success: function (success) {
                    success = JSON.parse(success);
                    var message = success["response"];
                    $("#assign_check_to_user_form").trigger("reset");
                    $.notific8(message, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000
                    });
                    load_checklist_table(setlist_ref)
                },
                error: function (error) {
                    console.log(error);
                    error = JSON.parse(error);
                    var message = error["response"];
                    console.log(message);
                    $("#assign_check_to_user_form").trigger("reset");
                    $.notific8(message, {
                        horizontalEdge: 'bottom',
                        verticalEdge: 'right',
                        zindex: 50000
                    });
                }
            });
        }
        );



        load_table();

        /*
         * checklist options modal show event
         */
        $('#checklistoptionModal').on('show.bs.modal', function (e) {
            var update_value = $(e.relatedTarget).data('check_list_value');
            var update_option_value = $(e.relatedTarget).data('check_option_value');


            setlist_ref = update_value;
            if (update_value !== 0) {
                $('<input>').attr({
                    type: 'hidden',
                    id: 'listref',
                    name: 'listref',
                    value: update_value
                }).appendTo('#checklist_option_form');
                load_checklist_table(update_value);
            }
            if (update_option_value !== 0) {
                get_checklist_option_by_id(update_option_value);
            }
        });

        /*
         * checklist modal show event
         */
        $('#checklistModal').on('show.bs.modal', function (e) {
            var update_value = $(e.relatedTarget).data('update_value');
            if (update_value !== 0) {
                $('<input>').attr({
                    type: 'hidden',
                    id: 'update_value',
                    name: 'update_value',
                    value: window.atob(update_value)
                }).appendTo('#checklist_form');
                get_clicklist_by_id(update_value);
            }
        });

        /*
         * assignToModel show event
         * call checklist options load function
         */
        $('#assignToModel').on('show.bs.modal', function (e) {
            get_checklist_options();
            get_users_options();
        });
        /*
         * checklist modal close event
         */
        $('#checklistModal').on('hide.bs.modal', function (e) {
            $("#update_value").remove();
            $("#checklist_form").trigger("reset");
        });

        /*
         * checklist options modal close event
         */
        $('#checklistoptionModal').on('hide.bs.modal', function (e) {
            $("#listref").remove();
            $("#checklist_option_form").trigger("reset");
        });


    });


    function get_checklist_options() {
        var url = "<?= base_url("hr_checklist_master/hr_checklist_option") ?>";
        $.ajax({
            url: url,
            method: "post"
        }).done(function (checklist) {
            var object = JSON.parse(checklist);
            console.log(object);
            $("#assign_check_list").empty();
            $("#assign_check_list").html(object['checklist_options']);

        }).fail(function (error) {
            $.notific8("Error while loading", {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });
    }

    function get_users_options() {
        var url = "<?= base_url("hr_checklist_master/hr_user_option") ?>";
        $.ajax({
            url: url,
            method: "post"
        }).done(function (checklist) {
            var object = JSON.parse(checklist);
            console.log(object);
            $("#assign_user_list").empty();
            $("#assign_user_list").html(object['userlist_options']);

        }).fail(function (error) {
            $.notific8("Error while loading", {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });
    }

    function get_clicklist_by_id(id) {
        var url = "<?= base_url() ?>hr_checklist_master/checklist_record";
        var data = {
            "list_id": id
        };
        $.ajax({
            url: url,
            data: data,
            method: "post"
        }).done(function (checklist) {
            var object = JSON.parse(checklist);
            $("input[name='check_title']").val(object['checklist'][0]["title"]);

        }).fail(function (error) {
            $.notific8(error, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });
    }

    function get_checklist_option_by_id(id) {
        var url = "<?= base_url() ?>hr_checklist_master/checklist_option_record";
        var data = {
            "list_id": id
        };
        $.ajax({
            url: url,
            data: data,
            method: "post"
        }).done(function (checklist) {
            var object = JSON.parse(checklist);
            $("input[name='check_title']").val(object['checklist'][0]["title"]);
            $("input[name='check_option']").val(object['checklist'][0]["option_type"]);
            $("input[name='check_description']").val(object['checklist'][0]["check_descp"]);

        }).fail(function (error) {
            $.notific8(error, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });
    }

    function checklist_delete_by(id) {
        var url = "<?= base_url("hr_checklist_master/remove") ?>";
        $.ajax({
            url: url,
            data: {update_value: window.btoa(id)},
            method: "post"
        }).done(function (success) {
            console.log(success);
            success = JSON.parse(success);
            var message = success["response"];
            console.log(message);
            load_table();
            $.notific8(message, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000
            });
        }).fail(function (error) {
            console.log(error);
            $.notific8(error, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });
    }

    function checklist_option_delete_by(id, ref) {
        checklist_delete_by(id);
        console.log(ref);
        load_checklist_table(window.btoa(ref));
    }

    function load_table() {
        var url = "<?= base_url("hr_checklist_master/list") ?>";
        $.ajax({
            url: url,
            method: "post"
        }).done(function (success) {
            success = JSON.parse(success);
            $('#check_list_tbody').empty();
            $('#check_list_tbody').append(success['checklist']);
            $('#check_list_table').DataTable();
        }).fail(function (error) {
            console.log(error);
            $.notific8(error, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });

    }

    function load_checklist_table(list_ref) {
        var url = "<?= base_url("hr_checklist_master/checklist_option_list") ?>";
        $.ajax({
            url: url,
            data: {"list_ref": list_ref},
            method: "post"
        }).done(function (success) {
            success = JSON.parse(success);
            console.log(success);
            $('#check_listoption_tbody').empty();
            $('#check_listoption_tbody').append(success['checklist']);
            $('#check_listoption_table').DataTable();
        }).fail(function (error) {
            console.log(error);
            $.notific8(error, {
                horizontalEdge: 'bottom',
                verticalEdge: 'right',
                zindex: 50000,
                theme: 'ruby'
            });
        });

    }
